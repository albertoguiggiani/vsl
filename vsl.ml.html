struct Stat {
pub mut:
	data   &Data
	name   string
	min_x  []f64
	max_x  []f64
	sum_x  []f64
	mean_x []f64
	sig_x  []f64
	del_x  []f64
	min_y  f64
	max_y  f64
	sum_y  f64
	mean_y f64
	sig_y  f64
	del_y  f64
}
